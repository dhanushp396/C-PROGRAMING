#include<stdio.h>
main()
{
	char ch,a,z,A,Z;
	
	printf("Enter a character : ");
	scanf("%c",&ch);
	
	if ((ch>='A') && (ch<='Z'))
	{
	printf("%c is an alphabet.",ch);
    }
    
	else if ((ch>='a') && (ch<='z'))
	{
	printf("%c is an alphabet.",ch);
    }
    
    else
    {
	printf("%c is not an alphabet.",ch);
    }
}
