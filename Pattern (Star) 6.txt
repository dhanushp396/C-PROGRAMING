#include<stdio.h>
main()
{
	int n,i,j,k;
	
	printf("Enter the number rows : ");
	scanf("%d",&n);
	
	for (i=1;i<=n;i++)
	{
		for (j=n-i;j>0;j--)
		{
			printf(" ");
		}
		
		for (k=1;k<=((2*i)-1);k++)
		{
			printf("*");
		}
		
		printf("\n");
	}
	
	for (i=n-1;i>0;i--)
	{
		for (k=1;k<=(n-i);k++)
		{
			printf(" ");
		}
		
		for (j=1;j<=((2*i)-1);j++)
		{
			printf("*");
		}
		
		printf("\n");
	}
}
