#include<stdio.h>
main()
{
	int num1,num2,sum;
	FILE *a,*b,*c;
	
	a = fopen("file1.txt","w");
	
	fprintf(a,"...Hello World...");
	
	fclose(a);
	
	b = fopen("file2.txt","r");
	
	fscanf(b,"%d %d",&num1,&num2);
	
	sum = num1+num2;
	
	fclose(b);
	
	c = fopen("file3.txt","w");
	
	fprintf(c,"The sum of %d and %d is %d.",num1,num2,sum);
	
	fclose(c);
}
